Развертывание на Serverless платформах

Ниже представлены примеры реализации serverless функций для различных платформ, которые помогут вам избежать проблем с CORS:

Пример универсальной serverless функции для OpenAI:

// Пример функции для Vercel, Netlify, и т.д.
export default async function handler(req, res) {
  // Установить CORS заголовки
  res.setHeader('Access-Control-Allow-Origin', '*');
  res.setHeader('Access-Control-Allow-Methods', 'POST, OPTIONS');
  res.setHeader('Access-Control-Allow-Headers', 'Content-Type, Authorization');
  
  // Обработать OPTIONS запрос для preflight
  if (req.method === 'OPTIONS') {
    return res.status(200).end();
  }
  
  if (req.method !== 'POST') {
    return res.status(405).json({ error: 'Method not allowed' });
  }
  
  try {
    const { provider, model, messages, apiKey } = req.body;
    
    if (!apiKey) {
      return res.status(400).json({ error: 'API key is required' });
    }
    
    if (!messages || !Array.isArray(messages)) {
      return res.status(400).json({ error: 'Valid messages array is required' });
    }
    
    // Разные эндпоинты в зависимости от провайдера
    let endpoint = '';
    let headers = { 'Content-Type': 'application/json' };
    let payload = {};
    
    switch (provider) {
      case 'openai':
        endpoint = 'https://api.openai.com/v1/chat/completions';
        headers.Authorization = `Bearer ${apiKey}`;
        payload = {
          model: model || 'gpt-4',
          messages: messages,
          temperature: 0.7
        };
        break;
        
      case 'google':
        endpoint = `https://generativelanguage.googleapis.com/v1beta/models/${model || 'gemini-pro'}:generateContent?key=${apiKey}`;
        payload = {
          contents: messages.map(msg => ({
            role: msg.role === 'user' ? 'USER' : 'MODEL',
            parts: [{ text: msg.content }]
          }))
        };
        break;
        
      // Добавьте другие провайдеры при необходимости
        
      default:
        return res.status(400).json({ error: 'Invalid provider' });
    }
    
    const response = await fetch(endpoint, {
      method: 'POST',
      headers: headers,
      body: JSON.stringify(payload)
    });
    
    const data = await response.json();
    
    if (!response.ok) {
      return res.status(response.status).json({ error: data.error || 'API request failed' });
    }
    
    return res.status(200).json(data);
  } catch (error) {
    console.error('Error:', error);
    return res.status(500).json({ error: 'Internal server error' });
  }
}

Как использовать:

  1. Выберите подходящую serverless платформу
  2. Создайте API endpoint используя пример выше
  3. Разверните ваше приложение на выбранной платформе
  4. Обновите ваш фронтенд, чтобы использовать ваш собственный API endpoint

Это позволит вам обойти ограничения CORS и безопасно использовать API ключи, не раскрывая их клиенту.
